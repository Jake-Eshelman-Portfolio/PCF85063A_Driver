47uint8_t *get_civic_time()
48{
49 const char *time_str = __TIME__;
50 const char *date_str = __DATE__;
51 static uint8_t time_array[7];
52
53 // Extract month
54 char month[DATE_MONTH_END + 1];
55 strncpy(month, date_str, DATE_MONTH_END);
56 month[DATE_MONTH_END] = '\0';
57 uint8_t decimal_month = find_month(month);
58 uint8_t bcd_month = convert_to_bcd(decimal_month);
59
60 // Extract date
61 char date[3];
62 int j = DATE_DAY_START, k = 0;
63 while (date_str[j] != ' ' && k < sizeof(date) - 1) {
64 date[k++] = date_str[j++];
65 }
66 date[k] = '\0';
67 uint8_t decimal_date = atoi(date);
68 uint8_t bcd_date = convert_to_bcd(decimal_date);
69
70 // Extract year
71 char year[YEAR_DIGITS + 1];
72 strncpy(year, &date_str[strlen(date_str) - YEAR_DIGITS], YEAR_DIGITS);
73 year[YEAR_DIGITS] = '\0';
74 uint8_t decimal_year = atoi(year);
75 uint8_t bcd_year = convert_to_bcd(decimal_year);
76
77 // Extract time
78 uint8_t hours = extract_time_component(time_str, TIME_HOURS_INDEX);
79 uint8_t minutes = extract_time_component(time_str, TIME_MINUTES_INDEX);
80 uint8_t seconds = extract_time_component(time_str, TIME_SECONDS_INDEX);
81
82 // Fill time array: sec, min, hr, day(1-31), weekday, month, year
83 time_array[SECONDS_INDEX] = seconds;
84 time_array[MINUTES_INDEX] = minutes;
85 time_array[HOURS_INDEX] = hours;
86 time_array[DATE_INDEX] = bcd_date;
87 time_array[WEEKDAY_INDEX] = 0; // Weekday is not set
88 time_array[MONTH_INDEX] = bcd_month;
89 time_array[YEAR_INDEX] = bcd_year;
90
91 return time_array;
92}

114rtc_error_t initialize_RTC(const uint8_t *time_array)
115{
116 if (time_array == NULL) {
117 LOG_ERR("Time array was null \n");
118 return RTC_ERROR_INVALID_PARAMETER;
119 }
120
121 // Validate alarm time values
122 if (time_array[SECONDS_INDEX] > 0x59 ||
123 time_array[MINUTES_INDEX] > 0x59 ||
124 time_array[HOURS_INDEX] > 0x23 ||
125 time_array[DATE_INDEX] > 0x31 ||
126 time_array[WEEKDAY_INDEX] != 0 ||
127 time_array[MONTH_INDEX] > 0x31 ||
128 time_array[YEAR_INDEX] == 0) {
129
130 LOG_ERR("Invalid time array values \n");
131 return RTC_ERROR_INVALID_PARAMETER;
132 }
133
134
135 pcf_85063A = DEVICE_DT_GET(DT_NODELABEL(i2c0));
136 if (!device_is_ready(pcf_85063A)) {
137 LOG_ERR("Error: i2c device is not ready\n");
138 return RTC_ERROR_DEVICE_SETUP;
139 }
140
141 if (!device_is_ready(int_gpio.port)) {
142 LOG_ERR("Error: interrupt GPIO device is not ready\n");
143 return RTC_ERROR_DEVICE_SETUP;
144 }
145
146 int ret = gpio_pin_configure_dt(&int_gpio, GPIO_INPUT);
147 if (ret != RTC_SUCCESS) {
148 LOG_ERR("Error %d: failed to configure interrupt pin\n", ret);
149 return RTC_ERROR_DEVICE_SETUP;
150 }
151
152 ret = gpio_pin_interrupt_configure_dt(&int_gpio, GPIO_INT_EDGE_TO_ACTIVE);
153 if (ret != RTC_SUCCESS) {
154 LOG_ERR("Error %d: failed to configure interrupt\n", ret);
155 return RTC_ERROR_DEVICE_SETUP;
156 }
157
158 gpio_init_callback(&gpio_cb, alarm_callback, BIT(int_gpio.pin));
159 gpio_add_callback(int_gpio.port, &gpio_cb);
160
161 // Write the time array to the registers
162 ret = i2c_burst_write(pcf_85063A, RTC_ADDRESS, RTC_TIME_REGISTER_ADDRESS, time_array,
163 RTC_TIME_REGISTER_SIZE);
164
165 if (ret != RTC_SUCCESS) {
166 LOG_ERR("Error %d: i2c burst write failed\n", ret);
167 return RTC_ERROR_I2C_WRITE;
168 }
169
170 return RTC_SUCCESS;
171}

182rtc_error_t read_register(uint8_t *read_buffer, const uint8_t size, const uint8_t start_address)
183{
184 // Input validation. Buffer is circular so reads beyond 18 are valid, but should not be allowed for clarity.
185 if (read_buffer == NULL) {
186 LOG_ERR("read_buffer was null \n");
187 return RTC_ERROR_INVALID_PARAMETER;
188 }
189
190 if (size <= 0 || size > RTC_REGISTER_SIZE) {
191 LOG_ERR("Invalid size: %d, size must be between 0 and 18 \n", size);
192 return RTC_ERROR_INVALID_PARAMETER;
193 }
194
195 if (start_address < 0 || start_address > RTC_REGISTER_SIZE ) {
196 LOG_ERR("Invalid start address: %d, start address must be between 0 and 17 \n", start_address);
197 return RTC_ERROR_INVALID_PARAMETER;
198 }
199
200 if((start_address + size) > RTC_REGISTER_SIZE) {
201 LOG_ERR("Invalid read range, outranged RTC registers");
202 return RTC_ERROR_INVALID_PARAMETER;
203 }
204
205 uint8_t ret = 0;
206 ret = i2c_burst_read(pcf_85063A, RTC_ADDRESS, start_address, read_buffer, size);
207
208 if (ret != RTC_SUCCESS) {
209 LOG_ERR("Error %d: burst read failed \n", ret);
210 return RTC_ERROR_I2C_WRITE;
211 }
212
213 for (int i = 0; i < size; i++) {
214 LOG_INF("Register[%d]: %02X \n", i, read_buffer[i]);
215 }
216
217 return RTC_SUCCESS;

283rtc_error_t set_alarm(const uint8_t *alarm_buffer, const size_t size)
284{
285 // Input validation
286 if (alarm_buffer == NULL) {
287 LOG_ERR("Alarm buffer is NULL \n");
288 return RTC_ERROR_INVALID_PARAMETER;
289 }
290
291 if (size != RTC_ALARM_REGISTER_SIZE) {
292 LOG_ERR("Invalid alarm buffer size. Expected %d, got %d \n",
293 RTC_ALARM_REGISTER_SIZE, size);
294 return RTC_ERROR_INVALID_PARAMETER;
295 }
296
297 // Validate alarm time values
298 if (alarm_buffer[SECONDS_INDEX] > 0x59 ||
299 alarm_buffer[MINUTES_INDEX] > 0x59 ||
300 alarm_buffer[HOURS_INDEX] > 0x23 ||
301 alarm_buffer[DATE_INDEX] > 0x31 ||
302 alarm_buffer[WEEKDAY_INDEX] != 0) {
303 LOG_ERR("Invalid alarm time values \n");
304 return RTC_ERROR_INVALID_PARAMETER;
305 }
306
307 // Set the control register
308 uint8_t control_2[ALARM_CONTROL_REGISTER], ret = 0;
309 control_2[0] = ENABLE_ALARM;
310 ret = write_register(control_2, sizeof(control_2), ALARM_CONTROL_REGISTER);
311 if (ret != RTC_SUCCESS) {
312 LOG_ERR("Error: %d. Failed to set alarm control register due to burst write failure \n", ret);
313 return RTC_ERROR_I2C_WRITE;
314 }
315
316 // Set the alarm register
317 ret = write_register(alarm_buffer, size, RTC_ALARM_REGISTER_ADDRESS);
318 if (ret != RTC_SUCCESS) {
319 LOG_ERR("Error: %d. Failed to set alarm time register due to burst write failure \n", ret);
320 return RTC_ERROR_I2C_WRITE;
321 }
322 k_work_init(&alarm_work, alarm_work_handler);
323 return RTC_SUCCESS;
324}
