77uint8_t *get_civic_time()
78{
79 const char *time_str = __TIME__;
80 const char *date_str = __DATE__;
81 static uint8_t time_array[7];
82
83 // Extract month
84 char month[DATE_MONTH_END + 1];
85 strncpy(month, date_str, DATE_MONTH_END);
86 month[DATE_MONTH_END] = '\0';
87 uint8_t decimal_month = find_month(month);
88 uint8_t bcd_month = convert_to_bcd(decimal_month);
89
90 // Extract date
91 char date[3];
92 int j = DATE_DAY_START, k = 0;
93 while (date_str[j] != ' ' && k < sizeof(date) - 1) {
94 date[k++] = date_str[j++];
95 }
96 date[k] = '\0';
97 uint8_t decimal_date = atoi(date);
98 uint8_t bcd_date = convert_to_bcd(decimal_date);
99
100 // Extract year
101 char year[YEAR_DIGITS + 1];
102 strncpy(year, &date_str[strlen(date_str) - YEAR_DIGITS], YEAR_DIGITS);
103 year[YEAR_DIGITS] = '\0';
104 uint8_t decimal_year = atoi(year);
105 uint8_t bcd_year = convert_to_bcd(decimal_year);
106
107 // Extract time
108 uint8_t hours = extract_time_component(time_str, TIME_HOURS_INDEX);
109 uint8_t minutes = extract_time_component(time_str, TIME_MINUTES_INDEX);
110 uint8_t seconds = extract_time_component(time_str, TIME_SECONDS_INDEX);
111
112 // Fill time array: sec, min, hr, day(1-31), weekday, month, year
113 time_array[SECONDS_INDEX] = seconds;
114 time_array[MINUTES_INDEX] = minutes;
115 time_array[HOURS_INDEX] = hours;
116 time_array[DATE_INDEX] = bcd_date;
117 time_array[WEEKDAY_INDEX] = 0; // Weekday is not set
118 time_array[MONTH_INDEX] = bcd_month;
119 time_array[YEAR_INDEX] = bcd_year;
120
121 return time_array;
122}

144rtc_error_t initialize_RTC(const uint8_t *time_array)
145{
146 if (time_array == NULL) {
147 LOG_ERR("Time array was null \n");
148 return RTC_ERROR_INVALID_PARAMETER;
149 }
150
151 // Validate alarm time values
152 if (time_array[SECONDS_INDEX] > 0x59 ||
153 time_array[MINUTES_INDEX] > 0x59 ||
154 time_array[HOURS_INDEX] > 0x23 ||
155 time_array[DATE_INDEX] > 0x31 ||
156 time_array[WEEKDAY_INDEX] != 0 ||
157 time_array[MONTH_INDEX] > 0x12 ||
158 time_array[YEAR_INDEX] == 0) {
159
160 LOG_ERR("Invalid time array values \n");
161 return RTC_ERROR_INVALID_PARAMETER;
162 }
163
164
165 pcf_85063A = DEVICE_DT_GET(DT_NODELABEL(i2c0));
166 if (!device_is_ready(pcf_85063A)) {
167 LOG_ERR("Error: i2c device is not ready\n");
168 return RTC_ERROR_DEVICE_SETUP;
169 }
170
171 if (!device_is_ready(int_gpio.port)) {
172 LOG_ERR("Error: interrupt GPIO device is not ready\n");
173 return RTC_ERROR_DEVICE_SETUP;
174 }
175
176 int ret = gpio_pin_configure_dt(&int_gpio, GPIO_INPUT);
177 if (ret != RTC_SUCCESS) {
178 LOG_ERR("Error %d: failed to configure interrupt pin\n", ret);
179 return RTC_ERROR_DEVICE_SETUP;
180 }
181
182 ret = gpio_pin_interrupt_configure_dt(&int_gpio, GPIO_INT_EDGE_TO_ACTIVE);
183 if (ret != RTC_SUCCESS) {
184 LOG_ERR("Error %d: failed to configure interrupt\n", ret);
185 return RTC_ERROR_DEVICE_SETUP;
186 }
187
188 gpio_init_callback(&gpio_cb, alarm_callback, BIT(int_gpio.pin));
189 gpio_add_callback(int_gpio.port, &gpio_cb);
190
191 // Write the time array to the registers
192 ret = i2c_burst_write(pcf_85063A, RTC_ADDRESS, RTC_TIME_REGISTER_ADDRESS, time_array,
193 RTC_TIME_REGISTER_SIZE);
194
195 if (ret != RTC_SUCCESS) {
196 LOG_ERR("Error %d: i2c burst write failed\n", ret);
197 return RTC_ERROR_I2C_WRITE;
198 }
199
200 return RTC_SUCCESS;
201}

212rtc_error_t read_register(uint8_t *read_buffer, const uint8_t size, const uint8_t start_address)
213{
214 // Input validation. Buffer is circular so reads beyond 18 are valid, but should not be allowed for clarity.
215 if (read_buffer == NULL) {
216 LOG_ERR("read_buffer was null \n");
217 return RTC_ERROR_INVALID_PARAMETER;
218 }
219
220 if (size <= 0 || size > RTC_REGISTER_SIZE) {
221 LOG_ERR("Invalid size: %d, size must be between 0 and 18 \n", size);
222 return RTC_ERROR_INVALID_PARAMETER;
223 }
224
225 if (start_address < 0 || start_address > RTC_REGISTER_SIZE ) {
226 LOG_ERR("Invalid start address: %d, start address must be between 0 and 17 \n", start_address);
227 return RTC_ERROR_INVALID_PARAMETER;
228 }
229
230 if((start_address + size) > RTC_REGISTER_SIZE) {
231 LOG_ERR("Invalid read range, outranged RTC registers");
232 return RTC_ERROR_INVALID_PARAMETER;
233 }
234
235 uint8_t ret = 0;
236 ret = i2c_burst_read(pcf_85063A, RTC_ADDRESS, start_address, read_buffer, size);
237
238 if (ret != RTC_SUCCESS) {
239 LOG_ERR("Error %d: burst read failed \n", ret);
240 return RTC_ERROR_I2C_WRITE;
241 }
242
243 for (int i = 0; i < size; i++) {
244 LOG_INF("Register[%d]: %02X \n", i, read_buffer[i]);
245 }
246
247 return RTC_SUCCESS;
248}

315rtc_error_t set_alarm(const uint8_t *alarm_buffer, const size_t size)
316{
317 // Input validation
318 if (alarm_buffer == NULL) {
319 LOG_ERR("Alarm buffer is NULL \n");
320 return RTC_ERROR_INVALID_PARAMETER;
321 }
322
323 if (size != RTC_ALARM_REGISTER_SIZE) {
324 LOG_ERR("Invalid alarm buffer size. Expected %d, got %d \n",
325 RTC_ALARM_REGISTER_SIZE, size);
326 return RTC_ERROR_INVALID_PARAMETER;
327 }
328
329 // Validate alarm time values
330 if (alarm_buffer[SECONDS_INDEX] > 0x59 ||
331 alarm_buffer[MINUTES_INDEX] > 0x59 ||
332 alarm_buffer[HOURS_INDEX] > 0x23 ||
333 alarm_buffer[DATE_INDEX] > 0x31 ||
334 alarm_buffer[WEEKDAY_INDEX] != 0) {
335 LOG_ERR("Invalid alarm time values \n");
336 return RTC_ERROR_INVALID_PARAMETER;
337 }
338
339 // Set the control register
340 uint8_t control_2[ALARM_CONTROL_REGISTER], ret = 0;
341 control_2[0] = ENABLE_ALARM;
342 ret = write_register(control_2, sizeof(control_2), ALARM_CONTROL_REGISTER);
343 if (ret != RTC_SUCCESS) {
344 LOG_ERR("Error: %d. Failed to set alarm control register due to burst write failure \n", ret);
345 return RTC_ERROR_I2C_WRITE;
346 }
347
348 // Set the alarm register
349 ret = write_register(alarm_buffer, size, RTC_ALARM_REGISTER_ADDRESS);
350 if (ret != RTC_SUCCESS) {
351 LOG_ERR("Error: %d. Failed to set alarm time register due to burst write failure \n", ret);
352 return RTC_ERROR_I2C_WRITE;
353 }
354 k_work_init(&alarm_work, alarm_work_handler);
355 return RTC_SUCCESS;
356}
